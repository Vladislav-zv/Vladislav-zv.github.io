dzis = new Date()--> twórzymy zmianą typu Date
dzis = dzis.getHours()--> bierzemy dane ze zmianej dzis(ilość godzin w tym czasie)
dzis = dzis.getMinutes()--> bierzemy dane ze zmianej dzis(ilość minut w tym czasie)
dzis = dzis.getMonth()+1--> bierzemy dane ze zmianej dzis(miesiąs+1, bo pierwszy m. jest 0 w tym czasie)
dzis = dzis.getDate()--> bierzemy dane ze zmianej dzis(dzisiejsza data)
dzis = dzis.getFullYear()--> bierzemy dane ze zmianej dzis(terażniejszy rok)
